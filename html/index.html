        ////Example - Hash
        
        var person = { name: "Jose", lastname: "Chacon" };
        var hash = new PJS.Hash();
        
        hash.add("16842461-2", person);

        console.debug(hash.get("16842461-2")); //Output: [Object person]

        hash.add(person, "Chile");

        console.debug(hash.get(person)); // Output: Chile
